350 __STATIC_INLINE void nrf_gpiote_task_enable(uint32_t idx)
351 {
352  uint32_t final_config = NRF_GPIOTE->CONFIG[idx] | GPIOTE_CONFIG_MODE_Task;
353 #ifdef NRF51
354  /* Workaround for the OUTINIT PAN. When nrf_gpiote_task_config() is called a glitch happens
355  on the GPIO if the GPIO in question is already assigned to GPIOTE and the pin is in the
356  correct state in GPIOTE but not in the OUT register. */
357  /* Configure channel to not existing, not connected to the pin, and configure as a tasks that will set it to proper level */
358  NRF_GPIOTE->CONFIG[idx] = final_config | (((31) << GPIOTE_CONFIG_PSEL_Pos) & GPIOTE_CONFIG_PORT_PIN_Msk);
359  __NOP();
360  __NOP();
361  __NOP();
362 #endif
363  NRF_GPIOTE->CONFIG[idx] = final_config;
364 }

371 __STATIC_INLINE void nrf_gpiote_task_configure(uint32_t idx, uint32_t pin,
372  nrf_gpiote_polarity_t polarity,
373  nrf_gpiote_outinit_t init_val)
374 {
375  NRF_GPIOTE->CONFIG[idx] &= ~(GPIOTE_CONFIG_PORT_PIN_Msk |
376  GPIOTE_CONFIG_POLARITY_Msk |
377  GPIOTE_CONFIG_OUTINIT_Msk);
378 
379  NRF_GPIOTE->CONFIG[idx] |= ((pin << GPIOTE_CONFIG_PSEL_Pos) & GPIOTE_CONFIG_PORT_PIN_Msk) |
380  ((polarity << GPIOTE_CONFIG_POLARITY_Pos) & GPIOTE_CONFIG_POLARITY_Msk) |
381  ((init_val << GPIOTE_CONFIG_OUTINIT_Pos) & GPIOTE_CONFIG_OUTINIT_Msk);
382 }

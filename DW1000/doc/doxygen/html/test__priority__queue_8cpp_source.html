38 // This function verifies the content of the priority queue to match the passed in messages
39 void VerifyPriorityQueueContent(Thread::PriorityQueue &aPriorityQueue, int aExpectedLength, ...)
40 {
41  va_list args;
42  Thread::Message *message;
43  Thread::Message *msgArg;
44  uint8_t curPriority = 0xff;
45  uint16_t msgCount, bufCount;
46 
47  // Check the `GetInfo`
48  aPriorityQueue.GetInfo(msgCount, bufCount);
49  VerifyOrQuit(msgCount == aExpectedLength, "GetInfo() result does not match expected len.\n");
50 
51  va_start(args, aExpectedLength);
52 
53  if (aExpectedLength == 0)
54  {
55  message = aPriorityQueue.GetHead();
56  VerifyOrQuit(message == NULL, "PriorityQueue is not empty when expected len is zero.\n");
57 
58  VerifyOrQuit(aPriorityQueue.GetHeadForPriority(0) == NULL, "GetHeadForPriority() non-NULL when empty\n");
59  VerifyOrQuit(aPriorityQueue.GetHeadForPriority(1) == NULL, "GetHeadForPriority() non-NULL when empty\n");
60  VerifyOrQuit(aPriorityQueue.GetHeadForPriority(2) == NULL, "GetHeadForPriority() non-NULL when empty\n");
61  VerifyOrQuit(aPriorityQueue.GetHeadForPriority(3) == NULL, "GetHeadForPriority() non-NULL when empty\n");
62  }
63  else
64  {
65  // Go through all messages in the queue and verify they match the passed-in messages
66  for (message = aPriorityQueue.GetHead(); message != NULL; message = message->GetNext())
67  {
68  VerifyOrQuit(aExpectedLength != 0, "PriorityQueue contains more entries than expected.\n");
69 
70  msgArg = va_arg(args, Thread::Message *);
71 
72  if (msgArg->GetPriority() != curPriority)
73  {
74  for (curPriority++; curPriority != msgArg->GetPriority(); curPriority++)
75  {
76  // Check the `GetHeadForPriority` is NULL if there are no expected message for this priority level.
77  VerifyOrQuit(
78  aPriorityQueue.GetHeadForPriority(curPriority) == NULL,
79  "PriorityQueue::GetHeadForPriority is non-NULL when no expected msg for this priority.\n"
80  );
81  }
82 
83  // Check the `GetHeadForPriority`.
84  VerifyOrQuit(
85  aPriorityQueue.GetHeadForPriority(curPriority) == msgArg,
86  "PriorityQueue::GetHeadForPriority failed.\n"
87  );
88  }
89 
90  // Check the queued message to match the one from argument list
91  VerifyOrQuit(msgArg == message, "PriorityQueue content does not match what is expected.\n");
92 
93  aExpectedLength--;
94  }
95 
96  VerifyOrQuit(aExpectedLength == 0, "PriorityQueue contains less entries than expected.\n");
97 
98  // Check the `GetHeadForPriority` is NULL if there are no expected message for any remaining priority level.
99  for (curPriority++; curPriority < 4; curPriority++)
100  {
101  VerifyOrQuit(
102  aPriorityQueue.GetHeadForPriority(curPriority) == NULL,
103  "PriorityQueue::GetHeadForPriority is non-NULL when no expected msg for this priority.\n"
104  );
105  }
106  }
107 
108  va_end(args);
109 }

111 // This function verifies the content of the all message queue to match the passed in messages
112 void VerifyAllMessagesContent(Thread::MessagePool &aMessagePool, int aExpectedLength, ...)
113 {
114  va_list args;
115  Thread::MessagePool::Iterator it;
116  Thread::Message *msgArg;
117 
118  va_start(args, aExpectedLength);
119 
120  if (aExpectedLength == 0)
121  {
122  VerifyOrQuit(aMessagePool.GetAllMessagesHead().IsEmpty(), "Head is not empty when expected len is zero.\n");
123  VerifyOrQuit(aMessagePool.GetAllMessagesTail().IsEmpty(), "Tail is not empty when expected len is zero.\n");
124  }
125  else
126  {
127  for (it = aMessagePool.GetAllMessagesHead(); !it.HasEnded() ; it.GoToNext())
128  {
129  VerifyOrQuit(aExpectedLength != 0, "AllMessagesQueue contains more entries than expected.\n");
130  msgArg = va_arg(args, Thread::Message *);
131  VerifyOrQuit(msgArg == it.GetMessage(), "AllMessagesQueue content does not match what is expected.\n");
132  aExpectedLength--;
133  }
134 
135  VerifyOrQuit(aExpectedLength == 0, "AllMessagesQueue contains less entries than expected.\n");
136  }
137 
138  va_end(args);
139 }

219  VerifyOrQuit(msgHigh[i] != NULL, "Message::New failed\n");

226  SuccessOrQuit(msgLow[i]->SetPriority(2), "Message:SetPriority failed\n");
227  msgVeryLow[i] = messagePool.New(Thread::Message::kTypeIp6, 0);
228  VerifyOrQuit(msgVeryLow[i] != NULL, "Message::New failed\n");
229  SuccessOrQuit(msgVeryLow[i]->SetPriority(3), "Message:SetPriority failed\n");
230  }
231 
232  // Check the failure case for `SetPriority` for invalid argument.
233  VerifyOrQuit(
234  msgHigh[2]->SetPriority(Thread::Message::kNumPriorities) == kThreadError_InvalidArgs,
235  "Message::SetPriority() with out of range value did not fail as expected.\n"
236  );
237 
238  // Check the `GetPriority()`

240  {
241  VerifyOrQuit(msgHigh[i]->GetPriority() == 0, "Message::GetPriority failed.\n");
242  VerifyOrQuit(msgMed[i]->GetPriority() == 1, "Message::GetPriority failed.\n");
243  VerifyOrQuit(msgLow[i]->GetPriority() == 2, "Message::GetPriority failed.\n");
244  VerifyOrQuit(msgVeryLow[i]->GetPriority() == 3, "Message::GetPriority failed.\n");
245  }

253  SuccessOrQuit(queue.Enqueue(*msgMed[1]), "PriorityQueue::Enqueue() failed.\n");
254  VerifyPriorityQueueContent(queue, 2, msgMed[0], msgMed[1]);
255  SuccessOrQuit(queue.Enqueue(*msgHigh[0]), "PriorityQueue::Enqueue() failed.\n");
256  VerifyPriorityQueueContent(queue, 3, msgHigh[0], msgMed[0], msgMed[1]);
257  SuccessOrQuit(queue.Enqueue(*msgHigh[1]), "PriorityQueue::Enqueue() failed.\n");
258  VerifyPriorityQueueContent(queue, 4, msgHigh[0], msgHigh[1], msgMed[0], msgMed[1]);
259  SuccessOrQuit(queue.Enqueue(*msgMed[2]), "PriorityQueue::Enqueue() failed.\n");
260  VerifyPriorityQueueContent(queue, 5, msgHigh[0], msgHigh[1], msgMed[0], msgMed[1], msgMed[2]);
261  SuccessOrQuit(queue.Enqueue(*msgVeryLow[0]), "PriorityQueue::Enqueue() failed.\n");
262  VerifyPriorityQueueContent(queue, 6, msgHigh[0], msgHigh[1], msgMed[0], msgMed[1], msgMed[2], msgVeryLow[0]);
263  SuccessOrQuit(queue.Enqueue(*msgLow[0]), "PriorityQueue::Enqueue() failed.\n");
264  VerifyPriorityQueueContent(queue, 7, msgHigh[0], msgHigh[1], msgMed[0], msgMed[1], msgMed[2], msgLow[0],
265  msgVeryLow[0]);
266 
267  // Check the MessagePool::Iterator methods.
268  VerifyOrQuit(it.IsEmpty(), "Iterator::IsEmpty() failed to return `true` for an empty iterator.\n");
269  VerifyOrQuit(it.GetNext().IsEmpty(), "Iterator::IsEmpty() failed to return `true` for an empty iterator.\n");
270  VerifyOrQuit(it.GetPrev().IsEmpty(), "Iterator::IsEmpty() failed to return `true` for an empty iterator.\n");
271  it.GoToNext();
272  VerifyOrQuit(it.IsEmpty(), "Iterator::IsEmpty() failed to return `true` for an empty iterator.\n");
273  it.GoToNext();
274  VerifyOrQuit(it.IsEmpty(), "Iterator::IsEmpty() failed to return `true` for an empty iterator.\n");
275

278  VerifyOrQuit(it.GetMessage() == msgHigh[0], "Iterator::GetMessage() failed.\n");
279  it = it.GetNext();
280  VerifyOrQuit(!it.IsEmpty(), "Iterator::IsEmpty() failed to return `false` when it is not empty.\n");
281  VerifyOrQuit(it.GetMessage() == msgHigh[1], "Iterator::GetNext() failed.\n");
282  it = it.GetPrev();
283  VerifyOrQuit(it.GetMessage() == msgHigh[0], "Iterator::GetPrev() failed.\n");
284  it = it.GetPrev();
285  VerifyOrQuit(it.HasEnded(), "Iterator::GetPrev() failed to return empty at head.\n");
286  it = messagePool.GetAllMessagesTail();
287  it = it.GetNext();
288  VerifyOrQuit(it.HasEnded(), "Iterator::GetNext() failed to return empty at tail.\n");
289 
290  // Check the AllMessage queue contents (should match the content of priority queue).
291  VerifyAllMessagesContent(messagePool, 7, msgHigh[0], msgHigh[1], msgMed[0], msgMed[1], msgMed[2], msgLow[0],
292  msgVeryLow[0]);
293  VerifyAllMessagesContentInReverse(messagePool, 7, msgVeryLow[0], msgLow[0], msgMed[2], msgMed[1], msgMed[0],
294  msgHigh[1], msgHigh[0]);
295 
296  // Remove messages in different order and check the content of queue in each step.
297  SuccessOrQuit(queue.Dequeue(*msgHigh[0]), "PriorityQueue::Dequeue() failed.\n");
298  VerifyPriorityQueueContent(queue, 6, msgHigh[1], msgMed[0], msgMed[1], msgMed[2], msgLow[0], msgVeryLow[0]);
299  SuccessOrQuit(queue.Dequeue(*msgMed[2]), "PriorityQueue::Dequeue() failed.\n");
300  VerifyPriorityQueueContent(queue, 5, msgHigh[1], msgMed[0], msgMed[1], msgLow[0], msgVeryLow[0]);
301  SuccessOrQuit(queue.Dequeue(*msgLow[0]), "PriorityQueue::Dequeue() failed.\n");
302  VerifyPriorityQueueContent(queue, 4, msgHigh[1], msgMed[0], msgMed[1], msgVeryLow[0]);
303  SuccessOrQuit(queue.Dequeue(*msgMed[1]), "PriorityQueue::Dequeue() failed.\n");

308  VerifyPriorityQueueContent(queue, 1, msgMed[0]);

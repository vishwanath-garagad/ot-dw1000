61 ThreadError TaskletScheduler::Post(Tasklet &aTasklet)
62 {
63  ThreadError error = kThreadError_None;
64 
65  VerifyOrExit(mTail != &aTasklet && aTasklet.mNext == NULL, error = kThreadError_Already);
66 
67  if (mTail == NULL)
68  {
69  mHead = &aTasklet;
70  mTail = &aTasklet;
71  otSignalTaskletPending(aTasklet.mScheduler.GetIp6()->GetInstance());
72  }
73  else
74  {
75  mTail->mNext = &aTasklet;
76  mTail = &aTasklet;
77  }
78 
79 exit:
80  return error;
81 }
